141 template <typename T> void Logger::log_constr(T *inst, std::vector<rttr::variant> args)
142 {
143  std::unique_lock<std::mutex> lk(mut);
144  for (auto &a : args) {
145  // this will rewove shared/weak/unique ptrs
146  if (a.get_type().is_wrapper()) {
147  a = a.extract_wrapped_value();
148  }
149  }
150  std::string class_name = rttr::type::get<T>().get_name().to_string();
151  constr[class_name].push_back({inst, std::move(args)});
152  operations.emplace_back(ConstrId{class_name, constr[class_name].size() - 1});
153 }

170 template <typename T> size_t Logger::get_id_from_ptr(T *ptr)
171 {
172  const std::string class_name = rttr::type::get<T>().get_name().to_string();
173  for (size_t i = 0; i < constr.at(class_name).size(); ++i) {
