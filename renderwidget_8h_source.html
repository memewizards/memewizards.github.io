38  void paint(QPainter *painter, const QStyleOptionViewItem &option, const QModelIndex &index) const override
39  {
40  if (index.column() == 1) {
41  painter->save();
42  QStyleOptionViewItem opt(option);
43  QStyle *style = opt.widget ? opt.widget->style() : QApplication::style();
44  const int textMargin = style->pixelMetric(QStyle::PM_FocusFrameHMargin) + 1;
45  style->drawPrimitive(QStyle::PE_PanelItemViewItem, &opt, painter, opt.widget);
46 
47  QFont font = painter->font();
48  font.setBold(true);
49  painter->setFont(font);
50  QRect r1 = option.rect;
51  r1.adjust(0, textMargin, 0, -textMargin);
52  int mid = int((r1.height() / 2));
53  r1.setBottom(r1.y() + mid);
54  QRect bounding;
55  painter->drawText(r1, Qt::AlignLeft | Qt::AlignTop, index.data().toString(), &bounding);
56  r1.moveTop(r1.bottom() - textMargin);
57  font.setBold(false);
58  painter->setFont(font);
59  painter->drawText(r1, Qt::AlignLeft | Qt::AlignTop, index.data(Qt::UserRole).toString());
60  int progress = index.data(Qt::UserRole + 3).toInt();
61  if (progress > 0 && progress < 100) {
62  // draw progress bar
63  QColor color = option.palette.alternateBase().color();
64  QColor fgColor = option.palette.text().color();
65  color.setAlpha(150);
66  fgColor.setAlpha(150);
67  painter->setBrush(QBrush(color));
68  painter->setPen(QPen(fgColor));
69  int width = qMin(200, r1.width() - 4);
70  QRect bgrect(r1.left() + 2, option.rect.bottom() - 6 - textMargin, width, 6);
71  painter->drawRoundedRect(bgrect, 3, 3);
72  painter->setBrush(QBrush(fgColor));
73  bgrect.adjust(2, 2, 0, -1);
74  painter->setPen(Qt::NoPen);
75  bgrect.setWidth((width - 2) * progress / 100);
76  painter->drawRect(bgrect);
77  } else {
78  r1.setBottom(opt.rect.bottom());
79  r1.setTop(r1.bottom() - mid);
80  painter->drawText(r1, Qt::AlignLeft | Qt::AlignBottom, index.data(Qt::UserRole + 5).toString());
81  }
82  painter->restore();
83  } else {
84  QStyledItemDelegate::paint(painter, option, index);
85  }
86  }
