96 template <class T> void DataQueue<T>::push(const T &item)
97 {
98  m_mutex.lock();
99  if (m_queue.size() == m_maxSize) {
100  switch (m_mode) {
101  case OverflowModeDiscardOldest:
102  m_queue.removeFirst();
103  m_queue.append(item);
104  break;
105  case OverflowModeDiscardNewest:
106  // This item is the newest so discard it and exit
107  break;
108  case OverflowModeWait:
109  m_notFullCondition.wait(&m_mutex);
110  m_queue.append(item);
111  break;
112  }
113  } else {
114  m_queue.append(item);
115  if (m_queue.size() == 1) {
116  m_notEmptyCondition.wakeOne();
117  }
118  }
119  m_mutex.unlock();
120 }
